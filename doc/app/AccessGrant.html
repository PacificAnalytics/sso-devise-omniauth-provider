def redirect_uri_for(redirect_uri,state)
  if redirect_uri =~ /\?/
    redirect_uri + "&code=#{code}&response_type=code&state=#{state}"
  else
    redirect_uri + "?code=#{code}&response_type=code&state=#{state}"
  end
end
